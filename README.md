   3. {
    "web": {
        "client_id": ".googleusercontent.com",
        "project_id": "Web ",
        "auth_uri": "https://accounts.google.comauth",
        "token_uri": "https:token",
        "client_secret": " ur secrety  ",
        "redirect_uris": [
            "http://localhost:5000/callback"
        ]
    }
}
